1. ئۇيغۇرچە تېكىستنىڭ قىسقىچە مەزمۇنىنى ئېلىش

بۇ فونكىسىيە PHP ئۆزى تەمىنلىگەن قىسىپ ئېلىش فونكىسىيەسى بىلەن پەرىقلىنىدىغان بۇلۇپ، PHP ئۆزى تەمىنلىگەن فونكىسىيەدە ئۇيغۇرچە (Unicode لۇق يېزىقلار) يېزىقنى قىسىپ ئالغاندا قالايمىقان كود چىقىپ قالىدۇ. لېكىن بۇ فوكىسىيەدە ھېچقانداق قالايمىغان كود كۆرۈنمەيدۇ. مەسىلەن: 
<?php
	$str = 'ئۇيغۇرچە تېكىستنىڭ قىسقىچە مەزمۇنىنى ئېلىش';
	echo mb_substr($str, 6,22);
	echo substr($str, 6,22);
	echo mid_ug($str, 6,15);
?>
نەتىجە:mb_substr فونكىسىيەنىڭ : غۇرچە تېكىس� بولىدۇ.
نەتىجە:substr فونكىسىيەنىڭمۇ نەتىجىسى mb_substr فونكىسىيە بىلەن ئوخشاش بولىدۇ.
نەتىجە:mid_ug فونكىسىيەنىڭ : چە تېكىستنىڭ قىسقى... بولىدۇ.
بۇ فونكىسىيەدە يەنە ئۇيغۇرچە تېكىست بىلەن ئېنگىلىزچە ھەرپ، ئەرەب رەقىمى، خەنزۇچە ۋە باشقا ھەرىپ بەلگىلەر بىلەن بىرىكتۈرۈپ يېزىلغان تېكىستنىمۇ بىرتەرەپ قىلالايدۇ بۇنداق بولغاندىمۇ يەنىلا ئوخشاشلا قالايمىقان كود بولمىغان توغرا نەتىجىگە ئېرىشەلەيمىز.

1.1 mid_ug فونكىسىيە بىلەن تۇنۇشۇش

mid_ug فونكىسىيەسىنىڭ ئىشلىتىش ئۇسۇلى PHP نىڭ ئۆزىنىڭ تەمىنلىگەن فونكىسىيەسى بىلەن ئاساسىي جەھەتتىن ئوخشاش بۇلۇپ، بۇ فونكىسىيەدە after پارامېتىر ئارتۇق. بۇ فونكىسىيەنىڭ گىرامماتىكىسى تۆۋەندىكىچە:

string mid_ug(string $string, [int $start = 0, [int $length = null, [string $after = '...']]] )

بۇ فونكىسىيە ئۇيغۇرچە تور بىكەتلەردە ۋە باشقا unicode لۇق مەزمۇننى قىسىپ ئالغان ئىشلىتىشكە باب كېلىدىغان بۇلۇپ، بۇ فونكىسىيەدە كىسىپ ئالغاندا ھېچقانداق قالايمىقان بەلگە چىقىپ تور بىكەتنىڭ سۈپىتىگە تەسىر كۆرسەتمەيدۇ. بۇ فونكىسىيەدە سانداندىن ئۇقۇلغان تىما مەزمۇنىنىڭ قىسقارتىلما مەزمۇنىنى كىسىپ ئېلىشقا تېخىمۇ باب كېلىدىغان بۇلۇپ. سىزنىڭ مەخسۇس قىسقا مەزمۇن ئىستونى تۇغۇزۇش مەسىلىڭىزنى ۋە PHP دا ئۇيغۇچە مەزمۇن قىسىپ ئېلىشى ھاجىتىڭىزدىن تۇلۇق چىقالايدۇ.

1.1.1 mid_ug فونكىسىيە پارامېتېر

$string پارامېتىر قىسىپ ئالماقچى بولغان مەزمۇننى كۆرسىتىدۇ.
$start مەزمۇننىڭ باشلانغان نۇقتىسى، قىممەت تىپى int بولىدۇ، كۆڭۈلدىكى قىممىتى 0 بۇلۇپ قىممەت بەلگىلىمىگەن ۋاقىتتا مەزمۇن بېشىدىن باشلاپ $length پارامېتىرنىڭ قىممىتىگىچە بولغان مەزمۇننى قىسىپ ئالىدۇ.
$length قىسىپ ئالماقچى بولغان ھەرپ ئۇزۇنلۇقى. قىممەت تىپى int بولىدۇ. كۆڭۈلدىكى قىممىتى NULL بۇلۇپ، قىممەت بەرمىگەندە $start دىن باشلاپ $string نىڭ ئاخىرغىچە بولغان مەزمۇننى قايتۇرىدۇ.
$after كىسىپ ئالغان مەزمۇننىڭ ئاخىرىغا ئۇلىنىپ كېلىدىغان مەزمۇننى كۆرسىتىدۇ. كۆڭۈلدىكى قىممىتى ... بۇلۇپ، ئەگەر $length پارامېتىر بەلگىلەنمىگەندە مەزمۇننىڭ ئاخىرىغا بۇ پارامېتىر قىممىتى ئۇلىنىپ كەلمەيدۇ.

1. مەن سانداندىن ئۇقۇپ ئالغان مەزمۇننىڭ ئىچىدىن 120 ھەرىپنى قىسقا مەزمۇن قىلىپ باش بەتتە كۆرسەتمەكچى ئىدىم، PHP نىڭ ئۆزىنىڭ تەمىنلىگەن فونكىسىيەسىدە قىلسام قالايمىقان كود پەيدا بۇلۇپ قالدى. قانداق قىلىپ ئويلىغىنىمدەك ئۇيغۇرچە مەزمۇننىڭ قىسقا مەزمۇنىنى ئالالايمەن ؟

مەن تەمىنلىگەن mid_ug فونكىسىيەنى ئىشلىتىڭ.

2. مىنىڭ بىر كىنو ساندان ئامبىرىم بار ئىدى مەن بۇنىڭدىكى كىنو قىسقىچە چۈشەندۈرۈلىشنىڭ باش قىسمىدىن باشلاپ 55 ھەرىپ نى قىسىپ ئېلىپ باش بەتتە كۆرسەتمەكچى ئىدىم ؟

2. ئۇيغۇر يېزىقى بىلەن ئۇيغۇر لاتىن يېزىقىنى ئۆز ئارا ئالماشتۇرۇش فونكىسىيەسى

بۇ فونكىسىيە ئۇيغۇر كونا يېزىقى بىلەن ئۇيغۇر لاتىن يېزىقىنى ئۆز ئارا ئۆلچەملىك ئالماشتۇرىدۇ. بۇ فونكىسىيە تور بىكەتلەردە رەسىم ۋە باشقا تىپتىكى ھۆججەت يۈكلىگەن ۋاقىتتا بەلگىلىگەن ئۇيغۇچە يېزىقنى ئۇيغۇر لاتىن ھەرپىگە ئالماشتۇرۇش ئۈچۈن ئىشلىتىشكا باب كېلىدۇ. ئۇيغۇرچە ھۆججەت ئىسىم بىلەن ساقلىغان ۋاقىتتا مۇلازىمېتىردا پەرىقلاندۈرۈلمەسلىك مەسىلىسى ۋە باشقا كۈتۈلمىگەن ئاۋارىچىلىقلاردىن ساقلىنىش ئۈچۈن يۈكلىگەن ماتېرىيالنىڭ ئىسمىنى ئاپتۇماتىك ئۇيغۇر لاتىن ھەرپىگە ئالماشتۇرۇپ ئىشلىتىلىدۇ.

1.1.2 ug_to_uly فونكىسىيە گرامماتىكىسى

ug_to_uly فونكىسىيەسىنىڭ ئىشلىتىش ئۇسۇلى ناھايىتى ئاددىي بۇلۇپ ئالماشتۇرماقچى بولغان ئۇيغۇرچە مەزمۇننى $str پارامېتىرىغا بەرسەكلا ئۇيغۇرچە لاتىن يېزىقىغا ئېرىشەلەيمىز.

string ug_to_uly(string $str, [bool $type = true] )

2.1.2 ug_to_uly فونكىسىيە پارامېتېر

$str ئالماشتۇرماقچى بولغان مەزمۇن
$type بۇ پارامېتىرنىڭ true بولسا ئۇيغۇرچە لاتىن يېزىقىغا ئالماشتۇرىدۇ. false بولسا ئۇيغۇر لاتىن يېزىقىنى ئۇيغۇر كونا يېزىقىغا ئالماشتۇرىدۇ. بۇ پارامېتىرنىڭ كۆڭۈلدىكى قىممىتى false بولىدۇ.

1.2.2 مىسال

<?php
	$str = 'ئانا تىلىم ئۇيغۇر تىلى.';
	echo ug_to_uly($str);
?>
1. نەتىجە : ana tilim uyghur tili.

ئۇيغۇرچە كونا يېزىقنى لاتىن ھەرپكە ئۆزگەرتىش مەشغۇلاتى ۋاقتىنچە توغرا بولماسلىق سەۋەبى بىلەن تەمىنلەنمىدى.

3. ئەرەب رەقىمىنى ئۇيغۇرچە چوڭ يېزىلىشقا ئۆزگەرتىش

بۇ فونكىسىيە مالىيە تۈرىگە كىرىدىغان بۇلۇپ مالىيە خىزمىتى ۋە دوكۇمېنىت تەمىنلەشكە ئېھتىياجلىق تور بىكەت ياسىغۇچى پىروگراممىرلارىنىڭ ئىشلىتىشىگە باب كېلىدۇ.

1.1.3 NTU فونكىسىيە گرامماتىكىسى

NTU فونكىسىيەسىنىڭ ئىشلىتىش ئۇسۇلى ناھايىتى ئاددىي بۇلۇپ ئالماشتۇرماقچى بولغان ئەرەب رەقەمدىكى پۇل سوممىسىنى $num پارامېتىرىغا بەرسەكلا ئۇيغۇر كونا يېزىقىدىكى پۇل سوممىسىنىڭ چوڭ يېزىلىشىغا ئېرىشەلەيمىز.

string NTU(string $num, [bool $after = false] )

2.1.3 NTU فونكىسىيە پارامېتېر

$num ئالماشتۇرماقچى بولغان ئەرەب رەقىمىدىكى پۇل سوممىسى
$after بۇ پارامېتىرنىڭ true بولسا يۈەن خېتىنى نەتىجە ئاخىرىغا قۇشۇپ كۆرسىتىپ بىرىدۇ، ئەگەرfalse بولسا يۈەن خېتىنى قوشمايدۇ. بۇ پارامېتىرنىڭ كۆڭۈلدىكى قىممىتى false بولىدۇ.

بۇ فونكىسىيەنىڭ $num پارامېتىر قىممىتى چوقۇم ھەرپ بەلگە بۇلىشى كېرەك. ئەگەر ھەرپ بەلگە تىپى بولمىغاندا كومپىيۇتېرنىڭ بەلگىلىگەن ھېسابلاش ئۇزۇنلۇققا بارغاندا ھېسابلاش قىممىتىدا ئۆزگىرىش بۇلۇپ، بۇنى بىر نورمال مەنبە دەپ ھېسابلىمايدۇ شۇڭىلاشقا قىممەت بىر قەدەر چوڭ بولغاندا ھېسابلاشتا خاتالىق كۆرىلىشى مۇمكىن، شۇڭىلاشقا خاتالىقنىڭ ئالدىنى ئېلىش ئۈچۈن بۇ پارامېتىرنىڭ تىپى چۇقۇم ھەرپ بەلگە تىپى بولىشى كېرەك.

1.2.3 مىسال

<?php
	$mumber = '12345678';
	echo NTU($number);
	$number = '100000000000008';
	echo NTU($number);
	$number = '100100010001008';
	echo NTU($number, true);
	$number = '999999999999999.999';
	echo NTU($number, true);
	$number = '001002';
	echo NTU($number, true);
?>
